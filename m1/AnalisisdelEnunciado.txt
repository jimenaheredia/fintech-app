```javascript
export class Cliente {
  constructor(id, nombre, apellido, dni, email, password) {
    this.id = id;
    this.nombre = nombre;
    this.apellido = apellido;
    this.dni = dni;

    this.password = password;

    this.cuentas.push(cuenta);

```

### `/modules/Cuenta.js`

```javascript
export class Cuenta {
  constructor(idCuenta, cliente, saldoInicial = 0) {
    this.idCuenta = idCuenta;
    this.cliente = cliente;
    this.saldo = saldoInicial;
    this.movimientos = [];
  }

  depositar(monto) {
    this.saldo += monto;
    this.movimientos.push({ tipo: "DepÃ³sito", monto, fecha: new Date() });
  }

  retirar(monto) {
    if (monto > this.saldo) return "Fondos insuficientes";
    this.saldo -= monto;
    this.movimientos.push({ tipo: "Retiro", monto, fecha: new Date() });
  }

  consultarSaldo() {
    return this.saldo;
  }
}
```

### (Opcional) `/modules/Blockchain.js`

```javascript
import crypto from "crypto-js";

export class Bloque {
  constructor(index, timestamp, data, prevHash = "") {
    this.index = index;
    this.timestamp = timestamp;
    this.data = data;
    this.prevHash = prevHash;
    this.hash = this.calcularHash();
  }

  calcularHash() {
    return crypto.SHA256(
      this.index + this.timestamp + JSON.stringify(this.data) + this.prevHash
    ).toString();
  }
}
```
